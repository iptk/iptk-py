class RedisRunner(Runner):
    def __init__(self, arg):
        super(RedisRunner, self).__init__()
        self.arg = arg

    def redis_connect(self, redis_url, queue):
        """
        Connect this runner to a Redis queue on the server specified by the
        given Redis URL. The runner will pull dataset identifiers to process
        from the queue. If the runner is unable to process a dataset because of
        missing input datasets, it will re-insert the dataset at the end of the
        queue. Note that this method will synchronously wait for datasets in an
        endless loop.
        
        For documentation on the format of the redis_url parameter, see
        http://redis-py.readthedocs.io/en/latest/#redis.StrictRedis.from_url
        """
        r = redis.StrictRedis.from_url(redis_url, decode_responses=True)
        data = r.blpop(queue)
        print(data)
